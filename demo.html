var count1 = 0;
var timer1 = 
	$.timer(
		function() {
			count1++;
			$('.one').html(count1);
		},
		1000,
		true
	);


var count2 = 0;
var timer2 =
	$.timer().set({
		action: function() {
			count2++;
			$('.two').html(count2);				
		},
		time: 2000
	}).play();
